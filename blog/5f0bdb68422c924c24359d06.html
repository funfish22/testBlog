function textBlink(ele,cls1,cls2,times){
    $(ele).each(function(){        //抓取全部相同的 ele
        var $this=$(this);         //宣告 $this 指向自己($(this))
        var i=0,c="";              //初始化
        setInterval(function(){    //時間延遲函式，times 之後再次執行
            i++;                   //i++ 等於 i = i + 1
            c=i%2?cls1:cls2;       //三元運算子，當i能被整除，c 就等於 cls1，反之 c 就等於 cls2
            $this.css("color",c);  //改變顏色
            $this.find("a").css("color",c)}    //因為 a 連結，不會因為父層給予顏色就會改變，所以查看子曾是否有 a 連結，也一併給予顏色
            ,times) // 幾秒之後再次執行 setInterval，此數字為毫秒
        }
    )
};

<li class="flicker"><p><a href="#">閃爍</a></p></li>

textBlink($(".flicker"), "#fff", "red", 300);
